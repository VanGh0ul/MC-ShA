enum user_statuses {
  admin
  usual
}

table user_status {
  id int [pk, increment]
  name varchar(20)
}

table user {
  id int [pk, increment]
  name varchar(50)
  company_name varchar(50)
  email varchar(20)
  phone varchar(12)
  avatar blob(4200000)
  password varchar(16)
  status int [ref: > user_status.id]
  reg_date date
  
  indexes {
    name
    company_name
  }
}

enum product_statuses {
  in_stock
  not_in_stock
}

enum measure_units {
  g
  kg
  t
  ml
  l
  units
}

table product_status {
  id int [pk, increment]
  name varchar(30)
}

table measure_unit_type {
  id int [pk, increment]
  name varchar(30)
}


table product {
  id int [pk, increment]
  seller_id int [ref: > user.id]
  name varchar(50)
  about text
  status int [ref: > product_status.id]
  img blob(4200000)
  added_on date
  modified_on date
  price int
  measure_unit int [ref: > measure_unit_type.id]
  
  indexes {
    name
  }
}

table category {
  id int [pk, increment]
  name varchar(50)
}

table category_content {
  category_id int [ref: > category.id]
  product_id int [ref: > product.id]
}

enum order_statuses {
  accepted
  declined
  waiting
  editing
  delivered
  canceled
}

table order_status {
  id int [pk, increment]
  name varchar(50)
}

table buy_order {
  id int [pk, increment]
  buyer_id int [ref: > user.id]
  seller_id int [ref: > user.id]
  sent_on date
  deliver_date date
  deliver_address varchar(50)
  status int [ref: > order_status.id]
}

table order_content {
  order_id int [ref: > buy_order.id]
  product_id int [ref: > product.id]
  quantity int
  product_name varchar(50)
  product_price int
  product_measure_unit varchar(30)
}
